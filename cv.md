## Code Example

Example task: _Count the number of duplicates_

```javascript
function duplicateCount(text) {
  const lower = text.toLowerCase();
  const counts = {};
  for (let i of lower) {
    counts[i] = (counts[i] || 0) + 1;
  }

  let result = 0;
  for (let key in counts) {
    if (counts[key] > 1) {
      result++;
    }
  }

  return result;
}
```
